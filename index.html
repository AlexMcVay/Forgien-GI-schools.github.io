import pandas as pd from pyodide.http import pyfetch from js import document async def load_excel_file(): file_input = document.getElementById("fileInput") file = file_input.files.item(0) if not file: print("No file selected") return # Read the uploaded Excel file file_reader = await file.text() df = pd.read_excel(file_reader, sheet_name="Comparison_Tool_Data_Full") # Filter for foreign schools (not in the US) df = df[df["country"] != "United States"] # Convert numeric columns df["bah"] = pd.to_numeric(df["bah"], errors="coerce") df["tuition in state"] = pd.to_numeric(df["tuition in state"], errors="coerce") df["Net Balance"] = df["bah"] - df["tuition in state"] # Display in table table = document.getElementById("data-table") for _, row in df.iterrows(): tr = document.createElement("tr") tr.innerHTML = f"{row['institution']}{row['country']}${row['Net Balance']}" table.appendChild(tr) document.getElementById("loadData").addEventListener("click", lambda event: load_excel_file())
